<?php
	$file = fopen("generic-food.csv", "r");
	$content = "";
	
	echo "<h3>" . fgets($file) . "</h3>";
	do {

		$row = fgets($file);
		if (mb_strpos($row, 'Fruit') !== false) {
			$content = explode(',', $row);

			if (strpos($content[1], ")")) {
				echo "<div>" . $content[0] . $content[1] . "</div>";
			} else {
				echo "<div>$content[0]</div>";
			}
		}
		

	} while ($row);
	fclose($file);
?>
